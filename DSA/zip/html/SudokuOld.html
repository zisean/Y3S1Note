  int[][] board =

  {{8, 0, 0, 0, 0, 7, 0, 0, 0},

   {0, 6, 2, 0, 4, 0, 0, 0, 0},

   {0, 3, 0, 0, 0, 5, 7, 0, 0},

   {9, 0, 0, 0, 0, 2, 0, 4, 0},

   {5, 0, 0, 4, 0, 1, 0, 0, 9},

   {0, 4, 0, 9, 0, 0, 0, 0, 7},

   {0, 0, 1, 3, 0, 0, 0, 7, 0},

   {0, 0, 0, 0, 5, 0, 2, 8, 0},

   {0, 0, 0, 8, 0, 0, 0, 0, 3},

  };



  class Cell {

    int i;

    int j;



    Cell(int i, int j) {

      this.i = i;

      this.j = j;

    }

  }



  public java.util.ArrayList<Cell> getFreeCellList() {

    java.util.ArrayList<Cell> list = new java.util.ArrayList<Cell>();

    for (int i = 0; i < 9; i++)

      for (int j = 0; j < 9; j++)

        if (board[i][j] == 0)

          list.add(new Cell(i, j));



    return list;

  }



  java.util.ArrayList<Cell> list = new java.util.ArrayList<Cell>();



  public SudokuOld() {

    list = getFreeCellList();

    Cell firstCell = list.get(0);

    board[firstCell.i][firstCell.j] = 1;

    if (search(0))

      printBoard();

    else

      System.out.println("No solution");

  }



  public void printBoard() {

    for (int i = 0; i < 9; i++) {

      for (int j = 0; j < 9; j++)

        System.out.print(board[i][j] + " ");

      System.out.println();

    }

  }



  public boolean search(int k) {

    boolean found = false; // Solution found?
    while (!found) {

      Cell cell = list.get(k);

      int i = cell.i;

      int j = cell.j;

      int currentValue = board[i][j];



      if (isValid(i, j)) {

        if (k + 1 == list.size()) {

          found = true;

        }

        else { // Move to the next free cell
          Cell nextCell = list.get(k + 1);

          board[nextCell.i][nextCell.j] = 1; // Start with 1
          k = k + 1;

        }

      }

      else if (currentValue < 9) {

        board[i][j] = board[i][j] + 1;

      }

      else { // currentValue is 9, backtrack
        while (board[i][j] == 9) {

          board[i][j] = 0;

          if (k == 0) {

            return false;

          }

          k--; // Backtrack
          i = list.get(k).i;

          j = list.get(k).j;

        }



        board[i][j] =  board[i][j] + 1;

      }

    }



    return true;

  }



  public boolean isValid(int i, int j) {

    // Check whether currentValue is at the i's row
    int currentValue = board[i][j];



    for (int column = 0; column < 9; column++)

      if (column != j && board[i][column] == currentValue)

        return false;



    for (int row = 0; row < 9; row++)

      if (row != i && board[row][j] == currentValue)

        return false;



    for (int row = (i / 3) * 3; row < (i / 3) * 3 + 3; row++)

      for (int column = (j / 3) * 3; column < (j / 3) * 3 + 3; column++)

        if (row != i && column != j && board[row][column] == currentValue)

          return false;



    return true;

  }
