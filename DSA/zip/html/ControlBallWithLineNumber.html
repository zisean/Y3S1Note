1  import javax.swing.*;
  2  import java.awt.*;
  3  import java.awt.event.*;
  4  
  5  public class ControlBall extends JFrame {
  6    private JButton jbtEnlarge = new JButton("Enlarge");
  7    private JButton jbtShrink = new JButton("Shrink");
  8    private BallCanvas canvas = new BallCanvas();
  9    
 10    public ControlBall() {
 11      JPanel panel = new JPanel(); // Use the panel to group buttons
 12      panel.add(jbtEnlarge);
 13      panel.add(jbtShrink);
 14      
 15      this.add(canvas, BorderLayout.CENTER); // Add canvas to center
 16      this.add(panel, BorderLayout.SOUTH); // Add buttons to the frame
 17      
 18      jbtEnlarge.addActionListener(new ActionListener() {
 19        @Override
 20        public void actionPerformed(ActionEvent e) {
 21          canvas.enlarge();
 22        }
 23      });
 24  
 25      jbtShrink.addActionListener(new ActionListener() {
 26        @Override
 27        public void actionPerformed(ActionEvent e) {
 28          canvas.shrink();
 29        }
 30      });    
 31    }
 32    
 33    /** Main method */
 34    public static void main(String[] args) {
 35      JFrame frame = new ControlBall();
 36      frame.setTitle("ControlBall");
 37      frame.setLocationRelativeTo(null); // Center the frame
 38      frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
 39      frame.setSize(200, 200);
 40      frame.setVisible(true);
 41    }
 42    
 43    public static class BallCanvas extends JPanel {
 44       private int radius = 5; // Default ball radius
 45      
 46      /** Enlarge the ball */
 47      public void enlarge() {
 48        radius += 1; 
 49        repaint();
 50      }
 51      
 52      /** Shrink the ball */
 53      public void shrink() {
 54        radius -= 1; 
 55        repaint();
 56      }
 57      
 58      @Override
 59      protected void paintComponent(Graphics g) {
 60        super.paintComponent(g);
 61        g.drawOval(getWidth() / 2 - radius, getHeight() / 2 - radius, 
 62          2 * radius, 2 * radius);
 63      }
 64    }
 65  }
