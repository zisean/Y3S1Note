1  import java.awt.*;
  2  import javax.swing.*;

  4  public class ShowSpringLayout extends JApplet {
  5    public ShowSpringLayout() {
  6      SpringLayout springLayout = new SpringLayout();
  7      JPanel p1 = new JPanel(springLayout);
  8      JButton jbt1 = new JButton("Button 1");
  9      p1.add(jbt1);
 10  
 11      Spring spring = Spring.constant(0, 1000, 2000);
 12      springLayout.putConstraint(SpringLayout.WEST, jbt1, spring,
 13                                 SpringLayout.WEST, p1);
 14                               springLayout.putConstraint(SpringLayout.EAST, p1,
 15                                 Spring.sum(spring, spring), SpringLayout.EAST, jbt1);
 16      springLayout.putConstraint(SpringLayout.NORTH, jbt1, spring,
 17                                 SpringLayout.NORTH, p1);
 18      springLayout.putConstraint(SpringLayout.SOUTH, p1, spring,
 19                                 SpringLayout.SOUTH, jbt1);
 20  
 21      add(p1, BorderLayout.CENTER);
 22    }
 23  
 24    public static void main(String[] args) {
 25      ShowSpringLayout applet = new ShowSpringLayout();
 26      JFrame frame = new JFrame();
 27      //EXIT_ON_CLOSE == 3
 28      frame.setDefaultCloseOperation(3);
 29      frame.setTitle("ShowSpringLayout");
 30      frame.getContentPane().add(applet, BorderLayout.CENTER);
 31      applet.init();
 32      applet.start();
 33      frame.setSize(400,320);
 34      Dimension d = Toolkit.getDefaultToolkit().getScreenSize();
 35      frame.setLocation((d.width - frame.getSize().width) / 2,
 36        (d.height - frame.getSize().height) / 2);
 37      frame.setVisible(true);
 38    }
