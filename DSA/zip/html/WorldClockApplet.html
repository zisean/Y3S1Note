// WorldClockApplet.java: Displays an international clock
import java.awt.*;
import javax.swing.*;

public class WorldClockApplet extends JApplet {
  /** Construct the applet */
  public WorldClockApplet() {
    getContentPane().add(new WorldClockControl());
  }

  public static void main(String[] args) {
    WorldClockApplet applet = new WorldClockApplet();
    JFrame frame = new JFrame("WorldClockApplet");
    frame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    frame.getContentPane().add(applet, BorderLayout.CENTER);
    frame.setSize(400, 320);
    frame.setVisible(true);
  }
}
