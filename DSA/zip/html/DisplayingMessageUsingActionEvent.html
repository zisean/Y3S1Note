import javax.swing.*;
import java.awt.*;
import java.awt.event.*;

public class DisplayingMessageUsingActionEvent extends JApplet {
  public DisplayingMessageUsingActionEvent() {
    MovingMessageNew messagePanel = new MovingMessageNew();
    getContentPane().add(messagePanel);
    TickUsingActionEvent tick = new TickUsingActionEvent();
    tick.setTickInterval(100);
    tick.addActionListener(messagePanel);
  }
}

class MovingMessageNew extends JPanel implements ActionListener {
  private int xCoordinate = 20;
  private int yCoordinate = 20;

  public void paintComponent(Graphics g) {
    super.paintComponent(g);
    if (xCoordinate > getWidth())
      xCoordinate = -20;
    else
      xCoordinate += 5;
    g.drawString("I am moving", xCoordinate, yCoordinate);
  }

  public void actionPerformed(ActionEvent e) {
    repaint();
  }
}
