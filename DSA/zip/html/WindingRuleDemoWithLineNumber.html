1  import java.awt.*;
  2  import java.awt.geom.*;
  3  import javax.swing.*;
  4  
  5  public class WindingRuleDemo extends JApplet {
  6    public WindingRuleDemo() {
  7      add(new ShapePanel());
  8    }
  9  
 10    class ShapePanel extends JPanel {
 11      @Override
 12      protected void paintComponent(Graphics g) {
 13        super.paintComponent(g);
 14    
 15        Graphics2D g2d = (Graphics2D)g; // Create Graphics2D
 16    
 17        g2d.translate(10, 10); // Translate to a new origin     
 18        g2d.draw(createAPath()); // Create and draw a path
 19  
 20        g2d.translate(160, 0); // Translate to a new origin
 21        Path2D path2 = createAPath(); // Create a path
 22        path2.setWindingRule(Path2D.WIND_EVEN_ODD); // Set a new rule
 23        g2d.fill(path2); // Create and fill a path
 24        
 25        g2d.translate(160, 0); // Translate to a new origin
 26        Path2D path3 = createAPath(); // Create a path
 27        path3.setWindingRule(Path2D.WIND_NON_ZERO); // Set a new rule
 28        g2d.fill(path3); // Create and fill a path
 29      }
 30      
 31      private Path2D createAPath() {
 32        // Define the outer rectangle
 33        Path2D path = new Path2D.Double();
 34        path.moveTo(0, 0);
 35        path.lineTo(0, 100);
 36        path.lineTo(100, 100);
 37        path.lineTo(100, 0);
 38        path.lineTo(0, 0);
 39  
 40        // Define the inner rectangle
 41        path.moveTo(30, 30);
 42        path.lineTo(30, 70);
 43        path.lineTo(70, 70);
 44        path.lineTo(70, 30);
 45        path.lineTo(30, 30);
 46        
 47        return path;
 48      }
 49    }
