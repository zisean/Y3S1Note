public class InitializationBlockDemo {

  public static void main(String[] args) {

    new InitializationBlockDemo();

  }



  public InitializationBlockDemo() {

    new Parrot();

  }



  {

    System.out.println("(2) InitializationBlockDemo's instance " +

      "initialization block is invoked");

  }



  static {

    System.out.println("(1) InitializationBlockDemo's static " +

      "initialization block is invoked");

  }

}



class Parrot extends Bird {

  Parrot() {

    System.out.println("(8) Parrot's constructor is invoked");

  }



  {

    System.out.println("(7) Parrot's instance initialization block "

      + "is invoked");

  }



  static {

    System.out.println("(4) Parrot's static initialization block " +

      "is invoked");

  }

}



class Bird {

  Bird() {

    System.out.println("(6) Bird's constructor is invoked");

  }



  {

    System.out.println("(5) Bird's instance initialization block " +

      "is invoked");

  }



  static {

    System.out.println("(3) Bird's static initialization block " +

      "is invoked");

  }

}
